1) find all employees who are assigned to a department?
SELECT e.empid,e.empname
FROM employees e
INNER JOIN departments d
ON e.deptid = d.deptid;

2) find all employees even if they are not assigned to any department?
SELECT e.empid,e.empname
FROM employees e
LEFT JOIN departments d
ON e.deptid = d.deptid;

3) find all department that have no employees
SELECT d.deptid,d.deptname
FROM employees e
RIGHT JOIN departments d
ON e.deptid = d.deptid;
WHERE e.empid IS NULL;

4) find customers who have never placed an order 
SELECT c.custid,c.custname
FROM customers c
LEFT JOIN orders o
ON c.custid = o.custid;
WHERE o.custid IS NULL;

5) find customers along with their order 

SELECT c.custid,c.custname,o.orderid
