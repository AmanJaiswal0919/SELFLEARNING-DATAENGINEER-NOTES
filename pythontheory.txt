It means same method name but it shows a different behaviour

-What is Abstraction?

It means same method name and same parameter but in child class also called Runtime polymorphisms

Python does not support method overloading directly, but it can be achieved using default arguments

12)Check Armstrong Number
a)
def is_armstrong(num):
    power = len(str(num))
    total = sum(int(digit) ** power for digit in str(num))
    return total == num
# Example
print(is_armstrong(153))   # True
print(is_armstrong(123))   # False
b)
def is_armstrong(num):
    temp = num
    power = len(str(num))
    total = 0
    while temp > 0:
        digit = temp % 10
        total += digit ** power
        temp //= 10
    return total == num
# Example
print(is_armstrong(153))   # True
print(is_armstrong(123))   # False

13)Find First Non Repeating Character
a)
from collections import Counter

def first_non_repeating(s):
    freq = Counter(s)
    for ch in s:
        if freq[ch] == 1:
            return ch
    return None
# Example
print(first_non_repeating("swiss"))   # w
print(first_non_repeating("level"))   # v
b)
def first_non_repeating(s):
    freq = {}
    for ch in s:
        freq[ch] = freq.get(ch, 0) + 1
    for ch in s:
        if freq[ch] == 1:
            return ch
    return None
# Example
print(first_non_repeating("programming"))  # p

15)Check Prime Number
a)
def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, n):
        if n % i == 0:
            return False
    return True
# Example
print(is_prime(7))   # True
print(is_prime(10))  # False
b)
import math
def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(math.sqrt(n)) + 1):
        if n % i == 0:
            return False
    return True

# Example
print(is_prime(29))  # True
print(is_prime(100)) # False

def count_words(s):
    count = 0
    in_word = False
    for ch in s:
        if ch != " " and not in_word:
            count += 1
            in_word = True
        elif ch == " ":
            in_word = False
    return count
